public VoxelShape makeShape(){
	VoxelShape shape = Shapes.empty();
	shape = Shapes.join(shape, Shapes.box(0.0625, 0, 0.375, 0.1875, 1, 0.625), BooleanOp.OR);
	shape = Shapes.join(shape, Shapes.box(0.1875, 0, 0.4375, 0.25, 1, 0.5625), BooleanOp.OR);
	shape = Shapes.join(shape, Shapes.box(0, 0, 0.4375, 0.0625, 1, 0.5625), BooleanOp.OR);
	shape = Shapes.join(shape, Shapes.box(0, 0, 0.125, 1, 1, 0.25), BooleanOp.OR);

	return shape;
}
