public VoxelShape makeShape(){
	VoxelShape shape = Shapes.empty();
	shape = Shapes.join(shape, Shapes.box(0, 0, 0, 1, 0.0625, 1), BooleanOp.OR);
	shape = Shapes.join(shape, Shapes.box(0.0625, 0.0625, 0, 1, 0.1875, 0.9375), BooleanOp.OR);
	shape = Shapes.join(shape, Shapes.box(0, 0.1875, 0, 1, 0.25, 1), BooleanOp.OR);
	shape = Shapes.join(shape, Shapes.box(0, 0.25, 0, 1, 1, 0.125), BooleanOp.OR);

	return shape;
}
