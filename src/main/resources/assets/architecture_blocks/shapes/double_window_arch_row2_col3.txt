public VoxelShape makeShape(){
	VoxelShape shape = Shapes.empty();
	shape = Shapes.join(shape, Shapes.box(0, 0, 0, 1, 1, 0.75), BooleanOp.OR);
	shape = Shapes.join(shape, Shapes.box(0, 0.5, 0.75, 1, 1, 1), BooleanOp.OR);
	shape = Shapes.join(shape, Shapes.box(0, 0.4375, 0.75, 0.125, 0.5, 1), BooleanOp.OR);
	shape = Shapes.join(shape, Shapes.box(0.875, 0.4375, 0.75, 1, 0.5, 1), BooleanOp.OR);

	return shape;
}
