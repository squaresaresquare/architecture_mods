public VoxelShape makeShape(){
	VoxelShape shape = Shapes.empty();
	shape = Shapes.join(shape, Shapes.box(0.3125, 0, 0.375, 0.4375, 1, 0.625), BooleanOp.OR);
	shape = Shapes.join(shape, Shapes.box(0.4375, 0, 0.4375, 0.5, 1, 0.5625), BooleanOp.OR);
	shape = Shapes.join(shape, Shapes.box(0.25, 0, 0.4375, 0.3125, 1, 0.5625), BooleanOp.OR);
	shape = Shapes.join(shape, Shapes.box(0, 0, 0.125, 1, 1, 0.25), BooleanOp.OR);

	return shape;
}
